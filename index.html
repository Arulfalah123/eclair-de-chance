/** * GOOGLE APPS SCRIPT — Éclair de Chance * * CARA DEPLOY: * 1. Buka script.google.com → buat project baru * 2. Paste seluruh kode ini, ganti SPREADSHEET_ID * 3. Buat 2 sheet di Spreadsheet: * - Sheet "Pesanan" (kolom header baris 1: Waktu | Nama | Email | No WA | Alamat | Pesanan | Total) * - Sheet "Stok" (kolom header baris 1: Produk | S | M | L | XL | XXL) * Isi data mulai baris 2: * noir | 25 | 25 | 25 | 25 | 25 * blanc | 25 | 25 | 25 | 25 | 25 * 4. Deploy → New Deployment → Web App * - Execute as: Me * - Who has access: Anyone (anonymous) * 5. Klik Deploy → copy URL → paste ke GAS_URL di index.html.html * * PENTING: Setiap kali edit kode GAS, buat "New Deployment" baru (bukan manage existing), * lalu update URL-nya di HTML. */ const SPREADSHEET_ID = "1JnWO7gSp1dOrssef_8v72p9fNfno46ufZb-YoYrEAhc"; const SHEET_PESANAN = "Pesanan"; const SHEET_STOK = "Stok"; const SIZES = ["S", "M", "L", "XL", "XXL"]; // Helper: tambahkan CORS headers ke semua response function corsResponse(data) { return ContentService .createTextOutput(JSON.stringify(data)) .setMimeType(ContentService.MimeType.JSON); } // ---------------------------------------- // doGet: dipakai untuk AMBIL STOK dan juga // menerima data pesanan (workaround CORS) // ---------------------------------------- function doGet(e) { const action = e.parameter.action; // Ambil stok if (action === "getStok") { try { const ss = SpreadsheetApp.openById(SPREADSHEET_ID); const sheet = ss.getSheetByName(SHEET_STOK); const rows = sheet.getDataRange().getValues(); const stok = {}; for (let r = 1; r < rows.length; r++) { const id = rows[r][0]; if (!id) continue; stok[id] = {}; SIZES.forEach((size, i) => { stok[id][size] = Number(rows[r][i + 1]) || 0; }); } return corsResponse({ status: "ok", stok: stok }); } catch (err) { return corsResponse({ status: "error", stok: {}, message: err.message }); } } // Simpan pesanan via GET (workaround CORS — data dikirim sebagai JSON di-encode di param) if (action === "saveOrder") { try { const raw = e.parameter.payload; const data = JSON.parse(decodeURIComponent(raw)); const ss = SpreadsheetApp.openById(SPREADSHEET_ID); // 1. Simpan ke sheet Pesanan const sheetPesanan = ss.getSheetByName(SHEET_PESANAN); sheetPesanan.appendRow([ data.waktu, data.nama, data.email, data.noWa, data.alamat, data.pesanan, data.total ]); // 2. Kurangi stok const sheetStok = ss.getSheetByName(SHEET_STOK); const stokRows = sheetStok.getDataRange().getValues(); if (Array.isArray(data.detail)) { data.detail.forEach(item => { const sizeIndex = SIZES.indexOf(item.size); if (sizeIndex === -1) return; for (let r = 1; r < stokRows.length; r++) { if (stokRows[r][0] === item.productId) { const col = sizeIndex + 2; // kolom B = index 1 = col 2 const current = Number(stokRows[r][sizeIndex + 1]) || 0; if (current > 0) { sheetStok.getRange(r + 1, col).setValue(current - 1); stokRows[r][sizeIndex + 1] = current - 1; // update cache lokal } break; } } }); } return corsResponse({ status: "ok" }); } catch (err) { return corsResponse({ status: "error", message: err.message }); } } return corsResponse({ status: "unknown action" }); } // doPost tetap ada sebagai fallback (tidak dipakai dari browser karena CORS) function doPost(e) { return doGet(e); }
